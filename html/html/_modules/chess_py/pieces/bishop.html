[docs]    def possible_moves(self, position):
        """
        Returns all possible bishop moves.

        :type position: Board
        :rtype: list
        """
        rook = Rook(self.color, self.location)
        moves = []

        for fn in self.diag_fn:
            moves.extend(rook.direction_moves(fn, position))

        for move in moves:
            move.piece = self


        return moves
